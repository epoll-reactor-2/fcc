65static inline size_t hashmap_hash(size_t k)
66{
67 k ^= k << 21;
68 k ^= k >> 17;
69#if (SIZE_MAX > 0xFFFFFFFF)
70 k ^= k >> 35;
71 k ^= k >> 51;
72#endif
73 return k;
74}

78static inline void hashmap_put(hashmap_t* map, size_t key, size_t val)
79{
80 size_t hash = hashmap_hash(key);
81 size_t index;
82 for (size_t i=0; i<HASHMAP_MAX_PROBES; ++i) {
83 index = (hash + i) & map->size;
84
85 if (map->buckets[index].key == key) {
86 // The key is already used, change value
87 map->buckets[index].val = val;
88
89 if (!val) {
90 // Value = 0 means we can clear a bucket
91 // Rebalance colliding trailing entries
92 hashmap_rebalance(map, index);
93 map->entries--;
94 }
95 return;
96 } else if (!map->buckets[index].val && val) {
97 // Empty bucket found, the key is unused
98 map->entries++;
99 map->buckets[index].key = key;
100 map->buckets[index].val = val;
101 return;
102 }
103 }
104 // Near-key space is polluted with colliding entries, reallocate and rehash
105 // Puts the new entry as well to simplify the inlined function
106 if (val) hashmap_grow(map, key, val);
107}

109static inline size_t hashmap_get(const hashmap_t* map, size_t key)
110{
111 if (map->size == 0) return 0;
112 size_t hash = hashmap_hash(key);
113 size_t index;
114 for (size_t i=0; i<HASHMAP_MAX_PROBES; ++i) {
115 index = (hash + i) & map->size;
116 if (map->buckets[index].key == key || !map->buckets[index].val) {
117 return map->buckets[index].val;
118 }
119 }
120 return 0;
121}

123static inline void hashmap_remove(hashmap_t* map, size_t key)
124{
125 // Treat value zero as removed key
126 hashmap_put(map, key, 0);
127 if (map->entries < (map->size >> 3) && map->entries > 1024) {
128 hashmap_shrink(map);
